\begin{Spell}[
  level=Cantrip,
  time=1 Action,
  range=60 ft,
  components={V},
  duration=1 Round,
  school=Enchantment,
  attack=INT Save,
  effect=Psychic
]{Mind Sliver}
You try to temporarily sliver the mind of one creature you can see within range. The target must succeed on an Intelligence saving throw or take 1d6 Psychic damage and subtract 1d4 from the next saving throw it makes before the end of your next turn.

\textbf{Cantrip Upgrade.} The damage increases by 1d6 when you reach levels 5 (2d6), 11 (3d6), and 17 (4d6).

\begin{components}
V & \textit{Mens fragilis (MENS frah-GEE-liss)}
\end{components}
\end{Spell}
