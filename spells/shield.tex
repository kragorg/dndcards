\begin{Spell}[
  level=1st,
  time=1 Reaction\fmark,
  range=Self,
  components={V, S},
  duration=1 Round,
  school=Abjuration,
  attack=None,
  effect=Warding
]{Shield}
  An imperceptible barrier of magical force protects you. Until the start of your next turn, you have a +5 bonus to AC, including against the triggering attack, and you take no damage from Magic Missile.

\begin{components}
V & \textit{Protego (proh-TEH-goh)}
\end{components}

\ftext{Which you take when you are hit by an attack roll or targeted by the Magic Missile spell.}
\end{Spell}
