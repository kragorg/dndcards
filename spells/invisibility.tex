\begin{Spell}[
  tags=Concentration,
  level=2nd,
  time=1 Action,
  range=Touch,
  components={V, S, M},
  duration=1 Hour,
  school=Illusion,
  attack=None,
  effect=Invisible
]{Invisibility}

\textbf{Using a Higher-Level Spell Slot.} You can target one additional creature for each spell slot level above 2.

\begin{components}
V & \textit{Evanesco (eh-vah-NESS-koh)} \\
M & an eyelash in gum arabic
\end{components}
\end{Spell}
